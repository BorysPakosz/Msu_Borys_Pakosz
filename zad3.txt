SELECT 
    last_name,
    salary,
    DENSE_RANK() OVER (ORDER BY salary DESC) AS ranking
FROM employees;

SELECT 
    last_name, 
    salary, 
    DENSE_RANK() OVER (ORDER BY salary DESC) AS ranking,
    SUM(salary) OVER () AS calkowita_suma_pensji
FROM employees;

CREATE TABLE sales AS 
SELECT * FROM hr.sales;

CREATE TABLE products AS 
SELECT * FROM hr.products;

SELECT 
    e.last_name, 
    p.product_name,
    SUM(s.quantity * s.price) OVER (
        PARTITION BY e.employee_id 
        ORDER BY s.sale_date, s.sale_id
    ) AS skumulowana_sprzedaz,
    RANK() OVER (
        ORDER BY (s.quantity * s.price) DESC
    ) AS ranking_sprzedazy
FROM employees e
JOIN sales s ON e.employee_id = s.employee_id
JOIN products p ON s.product_id = p.product_id;
5.

SELECT 
    e.last_name, 
    p.product_name, 
    s.sale_date,
    s.price,
    COUNT(*) OVER (PARTITION BY s.product_id, s.sale_date) AS transakcje_dzien,
    ROUND(SUM(s.quantity * s.price) OVER (PARTITION BY s.product_id, s.sale_date), 2) AS suma_dzienna_prod,
    LAG(s.price) OVER (PARTITION BY s.product_id ORDER BY s.sale_date, s.sale_id) AS poprz_cena,
    LEAD(s.price) OVER (PARTITION BY s.product_id ORDER BY s.sale_date, s.sale_id) AS nast_cena
FROM employees e
JOIN sales s ON e.employee_id = s.employee_id
JOIN products p ON s.product_id = p.product_id
ORDER BY s.product_id, s.sale_date, s.sale_id;

SELECT 
    p.product_name, 
    s.price,
    s.sale_date,
    SUM(s.quantity * s.price) OVER (
        PARTITION BY s.product_id, TRUNC(s.sale_date, 'MM')
    ) AS suma_miesieczna_produktu,
    SUM(s.quantity * s.price) OVER (
        PARTITION BY s.product_id, TRUNC(s.sale_date, 'MM') 
        ORDER BY s.sale_date, s.sale_id
    ) AS suma_rosnaca_w_miesiacu
FROM sales s
JOIN products p ON s.product_id = p.product_id
ORDER BY s.product_id, s.sale_date;

WITH sprzedaz_2022 AS (
    SELECT product_id, price, TO_CHAR(sale_date, 'MM-DD') AS dzien_miesiac
    FROM sales
    WHERE EXTRACT(YEAR FROM sale_date) = 2022
),
sprzedaz_2023 AS (
    SELECT product_id, price, TO_CHAR(sale_date, 'MM-DD') AS dzien_miesiac
    FROM sales
    WHERE EXTRACT(YEAR FROM sale_date) = 2023
)
SELECT 
    p.product_name,
    p.product_category,
    s22.dzien_miesiac AS dzien_sprzedazy,
    s22.price AS cena_2022,
    s23.price AS cena_2023,
    (s23.price - s22.price) AS roznica_cen
FROM sprzedaz_2022 s22
JOIN sprzedaz_2023 s23 
    ON s22.product_id = s23.product_id 
    AND s22.dzien_miesiac = s23.dzien_miesiac
JOIN products p 
    ON s22.product_id = p.product_id
ORDER BY p.product_category, p.product_name, s22.dzien_miesiac;

SELECT
    p.product_category,
    p.product_name,
    s.price,
    MIN(s.price) OVER (PARTITION BY p.product_category) AS min_cena_w_kategorii,
    MAX(s.price) OVER (PARTITION BY p.product_category) AS max_cena_w_kategorii,
    MAX(s.price) OVER (PARTITION BY p.product_category) - MIN(s.price) OVER 
    (PARTITION BY p.product_category) AS roznica_cen
FROM sales s
JOIN products p ON s.product_id = p.product_id
ORDER BY p.product_category, s.price;

9.
SELECT
    p.product_name,
    s.sale_date,
    s.price AS biezaca_cena,
    ROUND(AVG(s.price) OVER(
        PARTITION BY s.product_id
        ORDER BY s.sale_date, s.sale_id
        ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING
    ),2) AS srednia_kroczaca
FROM sales s
JOIN products p ON s.product_id = p.product_id
ORDER BY p.product_name, s.sale_date, s.sale_id;

10.

SELECT
    p.product_name,
    p.product_category,
    s.price,
    RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS ranking_zwykly,
    ROW_NUMBER() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS numer_wiersza,
    DENSE_RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS ranking_gesty
FROM sales s
JOIN products p ON s.product_id = p.product_id
ORDER BY p.product_category, s.price DESC;

SELECT
    e.last_name,
    p.product_name,
    SUM(s.quantity * s.price) OVER (
        PARTITION BY s.employee_id 
        ORDER BY s.sale_date, s.sale_id
    ) AS wartosc_rosnaca_sprzedazy,
    RANK() OVER (
        ORDER BY (s.quantity * s.price) DESC
    ) AS ranking_globalny
FROM sales s
JOIN employees e ON s.employee_id = e.employee_id
JOIN products p ON s.product_id = p.product_id
ORDER BY e.last_name, s.sale_date, s.sale_id;

    e.last_name, 
    j.job_title AS stanowisko
FROM employees e
JOIN jobs j ON e.job_id = j.job_id
WHERE EXISTS (
    SELECT 1 
    FROM sales s 
    WHERE s.employee_id = e.employee_id
);
